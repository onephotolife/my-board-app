Drawer デバッグ

1. http://localhost:3000 を開く

2. ハンバーガーメニューをクリック

3. ブラウザの開発者ツールで以下のコードを実行：

// MUIのDrawerを探す
const drawers = document.querySelectorAll('.MuiDrawer-root');
console.log('Drawer数:', drawers.length);

drawers.forEach((drawer, index) => {
    const paper = drawer.querySelector('.MuiPaper-root');
    if (paper) {
        const styles = window.getComputedStyle(paper);
        console.log(`\nDrawer ${index + 1}:`);
        console.log('  幅:', styles.width);
        console.log('  高さ:', styles.height);
        console.log('  位置:', styles.position);
        console.log('  左:', styles.left);
        console.log('  変形:', styles.transform);
        
        const rect = paper.getBoundingClientRect();
        console.log('  実際のサイズ:', rect.width, 'x', rect.height);
        console.log('  画面幅:', window.innerWidth);
        console.log('  全画面表示？:', rect.width === window.innerWidth ? '⚠️ はい' : '✅ いいえ');
    }
});

// SwipeableDrawerの要素を確認
const swipeableDrawer = document.querySelector('[class*="MuiSwipeableDrawer"]');
if (swipeableDrawer) {
    console.log('\nSwipeableDrawer found');
    const paper = swipeableDrawer.querySelector('.MuiPaper-root');
    if (paper) {
        console.log('Paper width attribute:', paper.style.width);
        console.log('Paper computed width:', window.getComputedStyle(paper).width);
    }
}
